    <script>
      const initialAirports = ["YBBN", "YSSY", "YMML", "YPPH", "NZAA", null];

      function getSeverityClass(code) {
        // Dummy logic to assign a color based on airport code
        const hash = code
          .split("")
          .reduce((acc, char) => acc + char.charCodeAt(0), 0);
        const colors = ["status-green", "status-orange", "status-red"];
        return colors[hash % 3];
      }

      function createWeatherContent(code, type = "METAR", horizon = 24) {
        // Dummy data with item and value pairs
        const data =
          type === "METAR"
            ? `Visibility: 10km\nWind: 120/12kt\nTemp: 26°C\nClouds: FEW020`
            : `Period: Next ${horizon} Hours\nWind: 140/10kt\nVisibility: 6.1km\nClouds: SCT030`;

        let horizonHtml = "";
        if (type === "TAF") {
          horizonHtml = `<select class="horizon-select" onchange="updateHorizon(this, '${code}')">`;
          for (let i = 8; i <= 30; i++) {
            horizonHtml += `<option value="${i}" ${i == horizon ? "selected" : ""}>${i}h</option>`;
          }
          horizonHtml += `</select>`;
        }

        const severityClass = getSeverityClass(code);

        return `
          <div class="card-header">
            <div class="airport">${code}</div>
            <div class="controls">
              ${horizonHtml}
              <div class="toggle-group">
                <button class="toggle-btn ${type === "METAR" ? "active" : ""}" onclick="toggleType(this, '${code}', 'METAR')">METAR</button>
                <button class="toggle-btn ${type === "TAF" ? "active" : ""}" onclick="toggleType(this, '${code}', 'TAF')">TAF</button>
              </div>
              <div class="delete-btn" onclick="revertToPlus(this)">x</div>
            </div>
          </div>
          <div class="label">${type} Data</div>
          <div class="weather">${data}</div>
          <div class="status-indicator ${severityClass}" title="Weather Status">⚠️</div>
        `;
      }

      function toggleType(btn, code, type) {
        const card = btn.closest(".card");
        card.innerHTML = createWeatherContent(code, type);
      }

      function updateHorizon(select, code) {
        const card = select.closest(".card");
        card.innerHTML = createWeatherContent(code, "TAF", select.value);
      }

      function revertToPlus(btn) {
        const card = btn.closest(".card");
        card.classList.add("add-card");
        card.onclick = () => addAirport(card);
        card.innerHTML = `<div class="plus">+</div>`;
      }

      function addAirport(cardElement) {
        const code = prompt("Enter ICAO Airport Code (e.g. YBBN):");
        if (!code) return;

        cardElement.classList.remove("add-card");
        cardElement.onclick = null;
        cardElement.innerHTML = createWeatherContent(code.toUpperCase());
      }

      function initDashboard() {
        const dashboard = document.getElementById("dashboard");
        dashboard.innerHTML = "";

        initialAirports.forEach((code) => {
          const card = document.createElement("div");
          if (code) {
            card.className = "card";
            card.innerHTML = createWeatherContent(code);
          } else {
            card.className = "card add-card";
            card.innerHTML = `<div class="plus">+</div>`;
            card.onclick = () => addAirport(card);
          }
          dashboard.appendChild(card);
        });
      }

      initDashboard();
    </script>
